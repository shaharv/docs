CReduce tips
------------

- Example command line:
  creduce ./run.sh md5-v1.c --timeout 30

- Example interestingness test:
 
  #!/bin/bash -exu
  
  # Verify with mainstream compilers, max warnings + ubsan + static analyzer
  gcc -O3  -fsanitize=undefined -fno-sanitize-recover -Wall -Wextra -Werror md5-v1.c && ./a.out
  gcc -O0  -fsanitize=undefined -fno-sanitize-recover -Wall -Wextra -Werror md5-v1.c && ./a.out
  clang-10 -fsanitize=undefined -fno-sanitize-recover=all -O1 -Wall -Wextra -Werror md5-v1.c && ./a.out
  clang-10 --analyze md5-v1.c 2>err.txt
  if [ -s err.txt ]; then
    echo "Static analyzer warnings found!"; exit 1
  fi

  # Verify with tools/options of choice
  $LLVM_BIN/clang --target=ppc -O1 -static md5-v1.c
  set +e
  ./a.out > out.txt 2<&1
  set -e
  # Use exact error string check to prevent false positives
  cat out.txt | grep "Assertion failed: result\[i\] == res1\[i\]"
  echo "OK."
